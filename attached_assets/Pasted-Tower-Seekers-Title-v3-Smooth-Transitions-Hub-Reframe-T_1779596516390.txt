The play() method currently swaps tracks instantly: stops synth, pauses old audioEl, immediately starts new. This is the jarring change you hear when going title â†’ /hub (skyforge â†’ hearthstone-tavern), or anywhere a different track loads.

4b. Replace _playFile(src) with a crossfading version
private _playFile(src: string) {
  this._removeBattleEndedHandler();

  // First-ever file play: instantiate primary, no fade needed.
  if (!this.audioEl) {
    this.audioEl = new Audio();
    this.audioEl.loop = true;
    this.audioEl.src = src;
    this.audioEl.volume = this._effectiveMusicVol();
    this.audioEl.play().catch(() => {});
    return;
  }

  // Same src already loaded & playing â†’ no-op (don't restart!)
  const currentSrc = this.audioEl.getAttribute("src") ?? "";
  const isSameSrc  = currentSrc.endsWith(src) || this.audioEl.src === src;
  if (isSameSrc && !this.audioEl.paused) {
    this.audioEl.volume = this._effectiveMusicVol();
    return;
  }

  // Different src â†’ crossfade.
  if (!this.audioElB) this.audioElB = new Audio();
  this.audioElB.loop = true;
  this.audioElB.src = src;
  this.audioElB.volume = 0;
  this.audioElB.play().catch(() => {});

  this._crossfade(this.audioEl, this.audioElB, this._effectiveMusicVol(), this.CROSSFADE_MS);

  // Swap roles so future calls treat audioElB as primary.
  const oldPrimary = this.audioEl;
  this.audioEl  = this.audioElB;
  this.audioElB = oldPrimary;
}
4c. Add _crossfade and _fadeOut helpers
private _crossfade(from: HTMLAudioElement, to: HTMLAudioElement, targetVol: number, durMs: number) {
  if (this.fadeRaf) cancelAnimationFrame(this.fadeRaf);
  const startVolFrom = from.volume;
  const startTime    = performance.now();
  const tick = (now: number) => {
    const t = Math.min(1, (now - startTime) / durMs);
    const eased = t * t * (3 - 2 * t); // smoothstep
    from.volume = startVolFrom * (1 - eased);
    to.volume   = targetVol * eased;
    if (t < 1) {
      this.fadeRaf = requestAnimationFrame(tick);
    } else {
      from.pause();
      from.currentTime = 0;
      this.fadeRaf = null;
    }
  };
  this.fadeRaf = requestAnimationFrame(tick);
}

private _fadeOut(el: HTMLAudioElement, durMs: number) {
  if (this.fadeRaf) cancelAnimationFrame(this.fadeRaf);
  const startVol = el.volume;
  const startTime = performance.now();
  const tick = (now: number) => {
    const t = Math.min(1, (now - startTime) / durMs);
    el.volume = startVol * (1 - t);
    if (t < 1) {
      this.fadeRaf = requestAnimationFrame(tick);
    } else {
      el.pause();
      el.currentTime = 0;
      this.fadeRaf = null;
    }
  };
  this.fadeRaf = requestAnimationFrame(tick);
}
